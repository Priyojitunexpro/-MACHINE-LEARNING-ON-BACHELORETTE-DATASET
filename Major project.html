#Takes the height in cm and returns it in ft and in
def to_inch(x):
    ft_raw = 0.0328084*x
    ft = int(ft_raw)
    rem = ft_raw-ft
    inches = round(rem*12,2)
    print('The average contestant is',ft,'feet and', inches,'inches tall.')
